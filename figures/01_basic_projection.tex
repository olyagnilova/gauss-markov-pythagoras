\tkzDefPoint(0,0){O}
\tkzDefPoint(4,0){X}
\tkzDefPoint(2,3){Y}

\draw[->] (O) -- (X) node[anchor=north]{$X$};
\draw[->, blue] (O) -- (Y) node[anchor=south, color=black]{$Y$};

\tkzMarkAngle[size=0.4](X,O,Y) % draw arc
\tkzLabelAngle[pos=0.6](X,O,Y){$\varphi$} % add label

\begin{pgfinterruptboundingbox} % remove blank space
\tkzDefLine[perpendicular=through Y](O,X);
\tkzGetPoint{hY}
\tkzDefPointBy[projection=onto X--O](hY);
\tkzGetPoint{yy}
\draw[dashed] (Y) --(yy);
\tkzMarkRightAngle(Y,yy,X);
\end{pgfinterruptboundingbox}

\draw[->, dashed, blue] (O) -- (yy) node[anchor=north, color=black]{$\hat Y$};

\end{tikzpicture}
