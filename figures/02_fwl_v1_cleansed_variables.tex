\documentclass{standalone}

\usepackage{tikz}   %TikZ is required for this to work.  Make sure this exists before the next line

\usepackage{tikz-3dplot} %requires 3dplot.sty to be in same directory, or in your LaTeX installation
\usepackage{tkz-euclide}
\usetkzobj{all}
\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work
\PreviewEnvironment{tikzpicture}
\setlength\PreviewBorder{2mm}

\begin{document}
\tdplotsetmaincoords{100}{110}
\begin{tikzpicture}[scale=5,tdplot_main_coords]

\coordinate (O) at (0,0,0);
\coordinate (x) at (1.75,1.5,0);
\coordinate (z) at (-1.75,0.5,0);
\coordinate (hx) at (0.63, 0.54, 0);
\coordinate (hz) at (-0.9625, 0.275, 0);
\coordinate (pz) at (-0.77, 0.22, 0);

\tdplotsetcoord{Y}{1}{50}{90}

\draw[->] (O) -- (x) node[anchor=south west]{$x$};
\draw[->] (O) -- (Y) node[anchor=south east]{$y$};
\draw[->] (O) -- (z) node[anchor=south east]{$z$};

\draw[->] (hz) -- (Y) node[anchor=north west]{$\tilde{y}$};
\tkzMarkRightAngle[size=0.03](Y,hz,z);
\draw[->] (O) -- (hz) node[anchor=north west]{$\hat\alpha z$};

\draw[->] (pz) -- (x) node[anchor=north west]{$\tilde{x}$};
\tkzMarkRightAngle[size=0.03](x,pz,O);
\draw[->] (O) -- (pz) node[anchor=north east]{$\hat\gamma z$};

\end{tikzpicture}
\end{document}
